---
title: "Create Folder"
api: "POST /api/folders"
description: "Create a new folder for organizing tasks"
---

## Request

<ParamField body="name" type="string" required>
  The name of the folder (max 100 characters)
</ParamField>

<ParamField body="description" type="string">
  Optional description of the folder
</ParamField>

<ParamField body="color" type="string">
  Color code for the folder (e.g., "#FF5733", max 7 characters)
</ParamField>

<ParamField body="icon" type="string">
  Icon identifier for the folder (max 50 characters)
</ParamField>

<ParamField body="parentFolderId" type="number">
  ID of the parent folder for nested folder structure
</ParamField>

<ParamField body="displayOrder" type="number" default="0">
  Order in which the folder should be displayed
</ParamField>

<ParamField body="tagIds" type="array">
  Array of tag IDs to associate with this folder
</ParamField>

## Response

<ResponseField name="folderId" type="number">
  Unique identifier for the folder
</ResponseField>

<ResponseField name="userId" type="number">
  ID of the user who owns the folder
</ResponseField>

<ResponseField name="tags" type="array">
  Array of tag objects associated with the folder
</ResponseField>

<ResponseField name="name" type="string">
  The name of the folder
</ResponseField>

<ResponseField name="description" type="string">
  Description of the folder
</ResponseField>

<ResponseField name="color" type="string">
  Color code for the folder
</ResponseField>

<ResponseField name="icon" type="string">
  Icon identifier for the folder
</ResponseField>

<ResponseField name="parentFolderId" type="number">
  ID of the parent folder
</ResponseField>

<ResponseField name="displayOrder" type="number">
  Display order of the folder
</ResponseField>

<ResponseField name="isArchived" type="boolean">
  Whether the folder is archived
</ResponseField>

<ResponseField name="createdAt" type="string">
  ISO 8601 timestamp of folder creation
</ResponseField>

<ResponseField name="updatedAt" type="string">
  ISO 8601 timestamp of last update
</ResponseField>

<RequestExample>
```bash cURL
curl -X POST https://api.taskpro.com/api/folders \
  -H "Authorization: Bearer YOUR_TOKEN" \
  -H "Content-Type: application/json" \

    "name": "Work Projects",
    "description": "All work-related projects and tasks",
    "color": "#4A90E2",
    "icon": "briefcase",
    "displayOrder": 1,
    "tagIds": [1, 2, 3]
  }'
```
</RequestExample>
