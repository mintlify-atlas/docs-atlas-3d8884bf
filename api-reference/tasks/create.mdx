---
title: 'Create Task'
api: 'POST /api/tasks'
description: 'Create a new task'
---

Create a new task with title, description, folder assignment, tags, priority, status, and due date.

## Request Body

<ParamField body="title" type="string" required>
  The title of the task
</ParamField>

<ParamField body="description" type="string">
  Optional description of the task
</ParamField>

<ParamField body="folderId" type="number">
  ID of the folder to assign the task to
</ParamField>

<ParamField body="tagIds" type="array">
  Array of tag IDs to associate with the task
</ParamField>

<ParamField body="durationTagId" type="number">
  ID of the duration tag for time estimation
</ParamField>

<ParamField body="priority" type="number" default="0">
  Priority level of the task (higher numbers = higher priority)
</ParamField>

<ParamField body="status" type="string" default="pending">
  Status of the task. Valid values: `pending`, `in_progress`, `completed`, `cancelled`
</ParamField>

<ParamField body="dueDate" type="string">
  Due date in ISO 8601 format (YYYY-MM-DD)
</ParamField>

<ParamField body="dueTime" type="string">
  Due time in ISO 8601 format (HH:mm:ss)
</ParamField>

<ParamField body="displayOrder" type="number" default="0">
  Display order for sorting tasks
</ParamField>

## Response

<ResponseField name="taskId" type="number">
  Unique identifier for the task
</ResponseField>

<ResponseField name="userId" type="number">
  ID of the user who owns the task
</ResponseField>

<ResponseField name="folderId" type="number">
  ID of the folder containing the task
</ResponseField>

<ResponseField name="tags" type="array">
  Array of tag objects associated with the task
</ResponseField>

<ResponseField name="title" type="string">
  Title of the task
</ResponseField>

<ResponseField name="description" type="string">
  Description of the task
</ResponseField>

<ResponseField name="durationTagId" type="number">
  ID of the duration tag
</ResponseField>

<ResponseField name="priority" type="number">
  Priority level of the task
</ResponseField>

<ResponseField name="status" type="string">
  Current status: `pending`, `in_progress`, `completed`, or `cancelled`
</ResponseField>

<ResponseField name="dueDate" type="string">
  Due date in ISO 8601 format
</ResponseField>

<ResponseField name="dueTime" type="string">
  Due time in ISO 8601 format
</ResponseField>

<ResponseField name="completedAt" type="string">
  Timestamp when the task was completed
</ResponseField>

<ResponseField name="displayOrder" type="number">
  Display order for sorting
</ResponseField>

<ResponseField name="isArchived" type="boolean">
  Whether the task is archived
</ResponseField>

<ResponseField name="createdAt" type="string">
  Timestamp when the task was created
</ResponseField>

<ResponseField name="updatedAt" type="string">
  Timestamp when the task was last updated
</ResponseField>
