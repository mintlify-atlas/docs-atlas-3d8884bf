---
title: 'List Tasks'
api: 'GET /api/tasks'
description: 'Get all tasks for the authenticated user'
---

Retrieve a list of all tasks belonging to the authenticated user, with optional filtering by folder, status, and archived state.

## Query Parameters

<ParamField query="folderId" type="number">
  Filter tasks by folder ID. Only returns tasks in the specified folder.
</ParamField>

<ParamField query="status" type="string">
  Filter tasks by status. Valid values: `pending`, `in_progress`, `completed`, `cancelled`
</ParamField>

<ParamField query="includeArchived" type="boolean" default="false">
  Include archived tasks in the results. By default, archived tasks are excluded.
</ParamField>

## Response

Returns an array of task objects with progress information:

<ResponseField name="taskId" type="number">
  Unique identifier for the task
</ResponseField>

<ResponseField name="userId" type="number">
  ID of the user who owns the task
</ResponseField>

<ResponseField name="folderId" type="number">
  ID of the folder containing the task
</ResponseField>

<ResponseField name="tags" type="array">
  Array of tag objects associated with the task
</ResponseField>

<ResponseField name="title" type="string">
  Title of the task
</ResponseField>

<ResponseField name="description" type="string">
  Description of the task
</ResponseField>

<ResponseField name="durationTagId" type="number">
  ID of the duration tag
</ResponseField>

<ResponseField name="priority" type="number">
  Priority level of the task
</ResponseField>

<ResponseField name="status" type="string">
  Current status: `pending`, `in_progress`, `completed`, or `cancelled`
</ResponseField>

<ResponseField name="dueDate" type="string">
  Due date in ISO 8601 format
</ResponseField>

<ResponseField name="dueTime" type="string">
  Due time in ISO 8601 format
</ResponseField>

<ResponseField name="completedAt" type="string">
  Timestamp when the task was completed
</ResponseField>

<ResponseField name="displayOrder" type="number">
  Display order for sorting
</ResponseField>

<ResponseField name="isArchived" type="boolean">
  Whether the task is archived
</ResponseField>

<ResponseField name="createdAt" type="string">
  Timestamp when the task was created
</ResponseField>

<ResponseField name="updatedAt" type="string">
  Timestamp when the task was last updated
</ResponseField>

<ResponseField name="totalSubtasks" type="number">
  Total number of subtasks
</ResponseField>

<ResponseField name="completedSubtasks" type="number">
  Number of completed subtasks
</ResponseField>

<ResponseField name="completionPercentage" type="number">
  Percentage of subtasks completed (0-100)
</ResponseField>

<RequestExample>

```bash cURL
# Get all tasks
curl https://api.taskpro.com/api/tasks \
  -H "Authorization: Bearer YOUR_API_KEY"

# Filter by folder
curl "https://api.taskpro.com/api/tasks?folderId=5" \
  -H "Authorization: Bearer YOUR_API_KEY"

# Filter by status
curl "https://api.taskpro.com/api/tasks?status=in_progress" \
  -H "Authorization: Bearer YOUR_API_KEY"

# Include archived tasks
curl "https://api.taskpro.com/api/tasks?includeArchived=true" \
  -H "Authorization: Bearer YOUR_API_KEY"

# Multiple filters
curl "https://api.taskpro.com/api/tasks?folderId=5&status=pending&includeArchived=false" \
  -H "Authorization: Bearer YOUR_API_KEY"
```

```javascript JavaScript
// Get all tasks
const response = await fetch('https://api.taskpro.com/api/tasks', {
  headers: {
    'Authorization': 'Bearer YOUR_API_KEY'
  }
});

const tasks = await response.json();

// With filters
const params = new URLSearchParams({
  folderId: '5',
  status: 'in_progress',
  includeArchived: 'false'
});

const filteredResponse = await fetch(
  `https://api.taskpro.com/api/tasks?${params}`,
  {
    headers: {
      'Authorization': 'Bearer YOUR_API_KEY'
    }
  }
);

const filteredTasks = await filteredResponse.json();
```

```python Python
import requests

# Get all tasks
response = requests.get(
    'https://api.taskpro.com/api/tasks',
    headers={'Authorization': 'Bearer YOUR_API_KEY'}
)

tasks = response.json()

# With filters
params = {
    'folderId': 5,
    'status': 'in_progress',
    'includeArchived': False
}

filtered_response = requests.get(
    'https://api.taskpro.com/api/tasks',
    headers={'Authorization': 'Bearer YOUR_API_KEY'},
    params=params
)

filtered_tasks = filtered_response.json()
```
