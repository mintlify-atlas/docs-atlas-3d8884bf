---
title: "Update Subtask"
api: "PUT /api/subtasks/{subtaskId}"
description: "Update an existing subtask"
---

## Request

Updates an existing subtask. All fields are optional - only provided fields will be updated.

### Headers

<ParamField header="Authorization" type="string" required>
  Bearer token for authentication
</ParamField>

### Path Parameters

<ParamField path="subtaskId" type="long" required>
  ID of the subtask to update
</ParamField>

### Body

<ParamField body="title" type="string">
  New title for the subtask (max 255 characters)
</ParamField>

<ParamField body="description" type="string">
  New description for the subtask
</ParamField>

<ParamField body="durationTagId" type="integer">
  New duration tag ID for time tracking
</ParamField>

<ParamField body="status" type="string">
  New status for the subtask. Valid values: `pending`, `in_progress`, `completed`
</ParamField>

<ParamField body="displayOrder" type="integer">
  New display order for the subtask
</ParamField>

<ParamField body="tagIds" type="array">
  New list of tag IDs to associate with this subtask (replaces existing tags)
</ParamField>

## Response

<ResponseField name="subtaskId" type="long">
  Unique identifier for the subtask
</ResponseField>

<ResponseField name="taskId" type="long">
  ID of the parent task
</ResponseField>

<ResponseField name="title" type="string">
  Title of the subtask
</ResponseField>

<ResponseField name="description" type="string">
  Description of the subtask
</ResponseField>

<ResponseField name="tags" type="array">
  Array of tag objects associated with the subtask
  <ResponseField name="tagId" type="long">
    Tag identifier
  </ResponseField>
  <ResponseField name="name" type="string">
    Tag name
  </ResponseField>
  <ResponseField name="color" type="string">
    Tag color
  </ResponseField>
</ResponseField>

<ResponseField name="durationTagId" type="integer">
  ID of the duration tag
</ResponseField>

<ResponseField name="status" type="string">
  Current status of the subtask
</ResponseField>

<ResponseField name="completedAt" type="string">
  ISO 8601 timestamp when the subtask was completed
</ResponseField>

<ResponseField name="displayOrder" type="integer">
  Display order of the subtask
</ResponseField>

<ResponseField name="createdAt" type="string">
  ISO 8601 timestamp when the subtask was created
</ResponseField>

<ResponseField name="updatedAt" type="string">
  ISO 8601 timestamp when the subtask was last updated
</ResponseField>

```bash cURL
  -H "Authorization: Bearer YOUR_TOKEN" \
  -H "Content-Type: application/json" \
  -d '{
    "status": "completed",
    "description": "All unit tests have been written and are passing"
  }'
```

```python Python
import requests

response = requests.put(
    'https://api.taskpro.com/api/subtasks/456',
    headers={
        'Authorization': 'Bearer YOUR_TOKEN',
        'Content-Type': 'application/json'
    },
    json={
        'status': 'completed',
        'description': 'All unit tests have been written and are passing'
    }
)

data = response.json()
```
