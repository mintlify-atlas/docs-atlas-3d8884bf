---
title: "Create Subtask"
api: "POST /api/subtasks"
description: "Create a new subtask under an existing task"
---

## Request

Creates a new subtask associated with a parent task. The subtask inherits access permissions from the parent task.

### Headers

<ParamField header="Authorization" type="string" required>
  Bearer token for authentication
</ParamField>

### Body

<ParamField body="taskId" type="long" required>
  ID of the parent task
</ParamField>

<ParamField body="title" type="string" required>
  Title of the subtask (max 255 characters)
</ParamField>

<ParamField body="description" type="string">
  Detailed description of the subtask
</ParamField>

<ParamField body="tagIds" type="array">
  List of tag IDs to associate with this subtask
</ParamField>

<ParamField body="durationTagId" type="integer">
  ID of the duration tag for time tracking
</ParamField>

<ParamField body="status" type="string" default="pending">
  Status of the subtask. Valid values: `pending`, `in_progress`, `completed`
</ParamField>

<ParamField body="displayOrder" type="integer" default="0">
  Order in which the subtask should be displayed
</ParamField>

## Response

<ResponseField name="subtaskId" type="long">
  Unique identifier for the created subtask
</ResponseField>

<ResponseField name="taskId" type="long">
  ID of the parent task
</ResponseField>

<ResponseField name="title" type="string">
  Title of the subtask
</ResponseField>

<ResponseField name="description" type="string">
  Description of the subtask
</ResponseField>

<ResponseField name="tags" type="array">
  Array of tag objects associated with the subtask
  <ResponseField name="tagId" type="long">
    Tag identifier
  </ResponseField>
  <ResponseField name="name" type="string">
    Tag name
  </ResponseField>
  <ResponseField name="color" type="string">
    Tag color
  </ResponseField>
</ResponseField>

<ResponseField name="durationTagId" type="integer">
  ID of the duration tag
</ResponseField>

<ResponseField name="status" type="string">
  Current status of the subtask
</ResponseField>

<ResponseField name="completedAt" type="string">
  ISO 8601 timestamp when the subtask was completed
</ResponseField>

<ResponseField name="displayOrder" type="integer">
  Display order of the subtask
</ResponseField>

<ResponseField name="createdAt" type="string">
  ISO 8601 timestamp when the subtask was created
</ResponseField>

<ResponseField name="updatedAt" type="string">
  ISO 8601 timestamp when the subtask was last updated
</ResponseField>

    'https://api.taskpro.com/api/subtasks',

        'Content-Type': 'application/json'

        'title': 'Write unit tests',
        'description': 'Add comprehensive unit tests for the authentication module',
        'tagIds': [1, 5],
        'durationTagId': 2,
        'status': 'pending',
        'displayOrder': 1
    }
)

data = response.json()
